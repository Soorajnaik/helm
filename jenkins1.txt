pipeline {
    agent {label 'EKS-node'}

    stages {
        stage('Clone') {
            steps {
                git branch: 'main', url: 'https://github.com/Soorajnaik/helm.git'
            }
        }
        stage('BuildingImage') {
            steps {
                sh 'pwd'
            }
        }
        stage('Helm_install_or_upgrade') {
            steps {
                sh 'helm upgrade --install sooraj-jenkis-shell client-app --namespace sooraj'
            }
        }

    }
}
